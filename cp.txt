func create(width, height){
    this.height = height;
    this.width = width;
    return this;
}

func area(){
    return this.width * this.height;
}

var rectangle = {
    create: create,
    area: area
};

var rect = rectangle.create(5,10);
println(rect.area());

var square = $_create(rectangle);

func create(side){
    return rectangle.create(side, side);
}

square.create = create;

var sq = square.create(5);
println(sq.area());
